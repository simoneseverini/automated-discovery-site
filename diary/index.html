Regarding the Shannon capacity $\Theta(G) = \lim_{d \to \infty} (\alpha(G^d))^{1/d}$, this paper improves the lower bound for $C_7$ by finding an independent set of size 367 in the fifth power. This yields:

$$ \Theta(C_7) \geq \sqrt[5]{367} \approx 3.2578 $$

See arXiv:1808.07438 for details.

View Construction Algorithm (Python)

This script uses an iterated greedy approach with Metropolis-Hastings inspired perturbation to find large independent sets in the 5th strong product power of $C_7$. 

import random
import time
import itertools

# Global cache to persist neighbors across function calls
_NEIGHBORS_CACHE = None

def to_index(t):
    """Convert a 5-tuple coordinate (base 7) to a scalar index."""
    return t[0] + 7 * (t[1] + 7 * (t[2] + 7 * (t[3] + 7 * t[4])))

def to_tuple(idx):
    """Convert a scalar index back to a 5-tuple coordinate."""
    x0 = idx % 7
    idx //= 7
    x1 = idx % 7
    idx //= 7
    x2 = idx % 7
    idx //= 7
    x3 = idx % 7
    x4 = idx // 7
    return (x0, x1, x2, x3, x4)

def find_independent_set_c7_5(time_limit_seconds, starting_set):
    """
    Search for a Maximum Independent Set in C7^5 using Iterated Greedy optimization.
    """
    start_time = time.time()
    N = 7**5  # 16807 vertices

    # --- Lazy Neighbor Initialization ---
    # Vertices u, v are connected if for all coords i: (u_i - v_i) % 7 in {-1, 0, 1}
    global _NEIGHBORS_CACHE
    if _NEIGHBORS_CACHE is None:
        diffs = [d for d in itertools.product((-1, 0, 1), repeat=5) if d != (0, 0, 0, 0, 0)]
        _NEIGHBORS_CACHE = [[] for _ in range(N)]
        for idx in range(N):
            t = to_tuple(idx)
            neighs = []
            for d in diffs:
                nt = (
                    (t[0] + d[0]) % 7,
                    (t[1] + d[1]) % 7,
                    (t[2] + d[2]) % 7,
                    (t[3] + d[3]) % 7,
                    (t[4] + d[4]) % 7
                )
                neighs.append(to_index(nt))
            _NEIGHBORS_CACHE[idx] = neighs

    # --- Search Logic ---
    conflicts = [0] * N
    current_set_s = set()
    
    def add_node(u):
        current_set_s.add(u)
        for v in _NEIGHBORS_CACHE[u]:
            conflicts[v] += 1

    def remove_node(u):
        current_set_s.remove(u)
        for v in _NEIGHBORS_CACHE[u]:
            conflicts[v] -= 1

    # Initialize with starting set
    for t in starting_set:
        idx = to_index(t)
        if idx not in current_set_s:
            add_node(idx)
            
    best_set = list(current_set_s)
    time_limit = start_time + time_limit_seconds * 0.98

    # Main Optimization Loop
    while time.time() < time_limit:
        # 1. Greedy Fill
        # Shuffle nodes to explore different greedy paths
        shuffled_nodes = list(range(N)) 
        random.shuffle(shuffled_nodes)
        
        for u in shuffled_nodes:
            # Add node if it has 0 neighbors in the current set
            if u not in current_set_s and conflicts[u] == 0:
                add_node(u)
        
        # Check for improvement
        if len(current_set_s) > len(best_set):
            best_set = list(current_set_s)

        # 2. Perturbation
        # Remove a random ~5% of nodes to escape local optima
        if current_set_s:
            perturb_size = max(1, len(current_set_s) // 20)
            to_remove = random.sample(list(current_set_s), min(perturb_size, len(current_set_s)))
            for r in to_remove:
                remove_node(r)

    return [to_tuple(idx) for idx in best_set]
